# Creating Dataframe and Pareto Chart
df = pd.DataFrame({'Dec': [7908206,7928214,7878962,8031252,8388278,8520898,8476331,8774736,8312241,
7078416,5865369,5796661,6120564,6036936,6213415,6069707,8147480,6080994,6326005,6339933]})
df.index = ['2001','2002','2003', '2004','2005','2006','2007','2008','2009','2010','2011','2012','2013',
'2014','2015','2016','2017','2018','2019','2020']

df = df.sort_values(by='Dec',ascending=False)
df["cumpercentage"] = df["Dec"].cumsum()/df["Dec"].sum()*100

fig, ax = plt.subplots(figsize=(20,10))
ax.bar(df.index, df["Dec"], color="C7")
ax2 = ax.twinx()
ax2.plot(df.index, df["cumpercentage"], color="C1", marker="D", ms=7)
ax2.yaxis.set_major_formatter(PercentFormatter())

ax.tick_params(axis="y", colors="C7")
ax2.tick_params(axis="y", colors="C7")
plt.title('Liters of diesel exported Years 2001-2020 (In Billions)')
plt.ylabel("Percentage of Sales")
plt.show()
